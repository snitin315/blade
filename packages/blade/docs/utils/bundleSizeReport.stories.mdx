} from '~components/Table';

<Table data={BUNDLE_SIZE_DATA}>
  {(tableData) => (
    <>
      <TableHeader>
        <TableHeaderRow>
          <TableHeaderCell>Component</TableHeaderCell>

          <TableHeaderCell>Size Limit</TableHeaderCell>
          <TableHeaderCell>Passed</TableHeaderCell>
        </TableHeaderRow>

      <TableBody>

            <TableCell>{`${tableItem.size/1000} kB`}</TableCell>
            <TableCell>
              {`${tableItem.sizeLimit/1000} kB`}
            </TableCell>
            <TableCell>
             <Badge
                size="medium"
                color={
                  tableItem.status === 'Completed'

                    : tableItem.status === 'Failed'
                    ? 'negative'
                    : 'default'
                }
                >
                {tableItem.passed ? 'Passed' : 'Failed'}
                </Badge>
            </TableCell>
          </TableRow>
